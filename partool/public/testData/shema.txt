    nr character varying(8),
    rtx_type integer,
    r_p_msisdn character varying(64),
    o_p_msisdn character varying(64),
    route_msisdn character varying(64),
    third_party character varying(64),
    imsi character varying(32),
    imei character varying(32),
    callstart date,
    calldatetime character varying(14),
    uom_code character varying(2),
    duration real,
    rated_amount real,
    rated_date date,
    location character varying(64),
    call_reference character varying(64),
    in_flag integer,
    trunk_in character varying(16),
    trunk_og character varying(16),
    volume_up integer,
    volume_down integer,
    tap_sender character varying(5),
    tap_amount real,
    tap_tax real,
    tap_callstart date,
    th_geo_nr integer,
    blacklist integer,
    msc character varying(64),
    event_type integer
);
